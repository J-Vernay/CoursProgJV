22 DinoVec2* pOutSize)
23{
24 DinoDrawCall drawCall;
25 drawCall.textureName = "monogram-bitmap.png";
26 // Il y a au maximum un quad = deux triangles pour chaque octet du texte, et un quad pour le fond.
27 drawCall.vertices.reserve(6 + 6 * text.size());
28
29 // On laisse les 6 premiers vertices pour le fond, pour qu'il soit dessiné en premier.
30 // On donnera les positions à la fin, car on ne connait pas encore la taille du texte.
31 if (colorBackground.a > 0)
32 drawCall.vertices.resize(6);
33
34 // Dimensions en pixels des lettres dans la texture.
35 constexpr uint16_t QUAD_WIDTH = 6;
36 constexpr uint16_t QUAD_HEIGHT = 12;
37
38 // Pour suivre la position des caractères au fur et à mesure.
39 int32_t x = 0;
40 int32_t y = 0;
41
42 // Pour suivre la taille totale du texte.
43 int32_t xMax = 0;
44
45 for (unsigned char c : text) {
46 if (c == '\n') {
47 // Retour à la ligne
48 xMax = std::max(xMax, x);
49 x = 0;
50 y += QUAD_HEIGHT;
51 }
52 else if (c <= 31) {
53 // Caractères de contrôle, on ne les gère pas = on les ignore = on ne fait rien.
54 }
55 else {
56 // Caractères étendus, en dehors de l'ASCII.
57 // On ne les gère pas = on affiche un '?' en tant qu'erreur.
58 if (c >= 128)
59 c = '?';
60
61 int32_t xIdx = c % 16;
62 int32_t yIdx = c / 16;
63 uint16_t u = static_cast<uint16_t>(xIdx * QUAD_WIDTH);
64 uint16_t v = static_cast<uint16_t>(yIdx * QUAD_HEIGHT);
65
66 DinoVec2 posA = {static_cast<float>(x), static_cast<float>(y)};
67 DinoVec2 posB = {static_cast<float>(x + QUAD_WIDTH), static_cast<float>(y)};
68 DinoVec2 posC = {static_cast<float>(x), static_cast<float>(y + QUAD_HEIGHT)};
69 DinoVec2 posD = {static_cast<float>(x + QUAD_WIDTH), static_cast<float>(y + QUAD_HEIGHT)};
70 drawCall.vertices.emplace_back(posA, u, v, color);
71 drawCall.vertices.emplace_back(posB, u + QUAD_WIDTH, v, color);
72 drawCall.vertices.emplace_back(posC, u, v + QUAD_HEIGHT, color);
73 drawCall.vertices.emplace_back(posB, u + QUAD_WIDTH, v, color);
74 drawCall.vertices.emplace_back(posC, u, v + QUAD_HEIGHT, color);
75 drawCall.vertices.emplace_back(posD, u + QUAD_WIDTH, v + QUAD_HEIGHT, color);
76
77 x += QUAD_WIDTH;
78 }
79 }
80 xMax = std::max(xMax, x);
81
82 float width = static_cast<float>(xMax);
83 float height = static_cast<float>(y + QUAD_HEIGHT);
84
85 if (colorBackground.a > 0) {
86 // Les 6 premiers points ont été laissés libres pour le fond.
87 drawCall.vertices[0] = {{0, 0}, 0, 0, colorBackground};
88 drawCall.vertices[1] = {{width, 0}, 0, 0, colorBackground};
89 drawCall.vertices[2] = {{0, height}, 0, 0, colorBackground};
90 drawCall.vertices[3] = {{width, 0}, 0, 0, colorBackground};
91 drawCall.vertices[4] = {{0, height}, 0, 0, colorBackground};
92 drawCall.vertices[5] = {{width, height}, 0, 0, colorBackground};
93 }

95 *pOutSize = {width, height};
96

100DinoDrawCall Dino_CreateDrawCall_Polyline(std::span<DinoVec2 const> points, float width,
101 DinoColor color)
102{
103 DinoDrawCall drawCall;
104
105 // En bonus, pour plus d'infos sur l'algorithme, voir :
106 // https://jvernay.fr/en/blog/points-triangulation/
107
108 // On ne supporte pas les couleurs transparentes, désolé.
109 if (color.a == 0)
110 return drawCall; // Invisible.
111 color.a = 255;
112
113 if (points.size() <= 1)
114 return drawCall; // Degenerate case: cannot trace a line from 0 or 1 point.
115
116 // Only compute these constants once.
117 float halfWidth = width / 2;
118
119 size_t idxPoint = 0;
120
121 // First point.
122 DinoVec2 A = points[idxPoint];
123 idxPoint += 1;
124
125 DinoVec2 B;
126 float lenAB = 0;
127 while (idxPoint < points.size() && lenAB == 0) {
128 // Get second point, such that A and B are distinct.
129 B = points[idxPoint];
130 lenAB = hypotf(B.x - A.x, B.y - A.y);
131 idxPoint += 1;
132 }
133 if (lenAB == 0)
134 return {}; // Degenerate case: all points are identical.
135
136 for (; idxPoint <= points.size(); idxPoint += 1) {
137
138 // Get next point C such that B and C are distinct.
139 DinoVec2 C;
140 float lenBC;
141 if (idxPoint < points.size()) {
142 C = points[idxPoint];
143 lenBC = hypotf(C.x - B.x, C.y - B.y);
144 }
145 else {
146 // [AB] is the last segment to render, no join is needed at the end.
147 // Making C = A such that (AB) and (BC) are aligned, thus no join will be generated.
148 C = A;
149 lenBC = lenAB;
150 }
151 if (lenBC == 0)
152 continue; // Empty segment, do nothing.
153
154 // Compute quad for segment AB.
155
156 float xAB = B.x - A.x, yAB = B.y - A.y;
157 float xAA1 = halfWidth / lenAB * -yAB;
158 float yAA1 = halfWidth / lenAB * xAB;
159 DinoVec2 A1{A.x + xAA1, A.y + yAA1};
160 DinoVec2 A2{A.x - xAA1, A.y - yAA1};
161 DinoVec2 A1p{B.x + xAA1, B.y + yAA1};
162 DinoVec2 A2p{B.x - xAA1, B.y - yAA1};
163
164 // Encode quad for segment AB as two triangles.
165 drawCall.vertices.emplace_back(A1, 0, 0, color);
166 drawCall.vertices.emplace_back(A2, 0, 0, color);
167 drawCall.vertices.emplace_back(A1p, 0, 0, color);
168 drawCall.vertices.emplace_back(A2, 0, 0, color);
169 drawCall.vertices.emplace_back(A1p, 0, 0, color);
170 drawCall.vertices.emplace_back(A2p, 0, 0, color);
171
172 // Determine the ABC angle's orientation.
173
174 float xBC = C.x - B.x, yBC = C.y - B.y;
175 float zAB_BC = xAB * yBC - yAB * xBC;
176
177 // If zAB_BC == 0, A B C are aligned and no join is needed.
178 if (zAB_BC != 0) {
179 // Compute the endpoints of the next segment.
180
181 float xBB1 = halfWidth / lenBC * -yBC;
182 float yBB1 = halfWidth / lenBC * xBC;
183 DinoVec2 B1{B.x + xBB1, B.y + yBB1};
184 DinoVec2 B2{B.x - xBB1, B.y - yBB1};
185
186 // Generate Bevel join triangle.
187 if (zAB_BC < 0) {
188 drawCall.vertices.emplace_back(B, 0, 0, color);
189 drawCall.vertices.emplace_back(A1p, 0, 0, color);
190 drawCall.vertices.emplace_back(B1, 0, 0, color);
191 }
192 else {
193 drawCall.vertices.emplace_back(B, 0, 0, color);
194 drawCall.vertices.emplace_back(A2p, 0, 0, color);
195 drawCall.vertices.emplace_back(B2, 0, 0, color);
196 }
197 }
198 // Prepare for next segment.
199 A = B;
200 B = C;
201 lenAB = lenBC;
202 }
203
204 return drawCall;
205}
