35

40
41 for (DinoGamepadIdx gamepadIdx : DinoGamepadIdx_ALL) {
42 DinoGamepad gamepad{};
43 bool bSuccess = XDino_GetGamepad(gamepadIdx, gamepad);
44 if (!bSuccess)
45 continue;
46
47 if (gamepad.btn_down && !gamepad.btn_up)
48 scale /= 1.01;
49 if (gamepad.btn_up && !gamepad.btn_down)
50 scale *= 1.01;
51 if (gamepad.btn_left && !gamepad.btn_right)
52 rotation += 90.0 * deltaTime;
53 if (gamepad.btn_right && !gamepad.btn_left)

55
56 circlePos.x += gamepad.stick_left_x * CIRCLE_SPEED * deltaTime;
57 circlePos.y += gamepad.stick_left_y * CIRCLE_SPEED * deltaTime;
58 }
