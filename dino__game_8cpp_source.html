95void Dino_GameFrame(double timeSinceStart)
96{
97 // Prendre en compte le temps qui passe.
98
99 float deltaTime = static_cast<float>(timeSinceStart - g_lastTime);
100 g_lastTime = timeSinceStart;
101
102 // Gestion des entrées et mise à jour de la logique de jeu.
103
104 for (DinoGamepadIdx gamepadIdx : DinoGamepadIdx_ALL) {
105 DinoGamepad gamepad{};
106 bool bSuccess = XDino_GetGamepad(gamepadIdx, gamepad);
107 if (!bSuccess)
108 continue;
109
110 if (gamepad.btn_down && !gamepad.btn_up)
111 g_scale /= 1.01;
112 if (gamepad.btn_up && !gamepad.btn_down)
113 g_scale *= 1.01;
114 if (gamepad.btn_left && !gamepad.btn_right)
115 g_rotation += 90.0 * deltaTime;
116 if (gamepad.btn_right && !gamepad.btn_left)
117 g_rotation -= 90.0 * deltaTime;
118
119 g_circlePos.x += gamepad.stick_left_x * CIRCLE_SPEED * deltaTime;
120 g_circlePos.y += gamepad.stick_left_y * CIRCLE_SPEED * deltaTime;
121 }
122
123 // Affichage
124
125 constexpr DinoColor CLEAR_COLOR = {50, 50, 80, 255};
126
127 XDino_SetClearColor(CLEAR_COLOR);
128
129 // Dessin de la "polyligne"
130 XDino_Draw(vbufID_polyline, XDino_TEXID_WHITE);
131
132 // Si on veut une correspondance 1:1 entre pixels logiques et pixels à l'écran.
133 // DinoVec2 windowSize = XDino_GetWindowSize();
134 // XDino_SetRenderSize(windowSize);
135 DinoVec2 renderSize = XDino_GetRenderSize();
136
137 // Dessin de la texture centrale qu'on peut bouger.
138 DinoVec2 translation = {renderSize.x / 2, renderSize.y / 2};
139 double scale = g_scale * std::min(renderSize.x, renderSize.y) / 4;
140 XDino_Draw(vbufID_imageMilieu, texID_imageMilieu, translation, scale, g_rotation);
141
142 // Dessin du cercle que l'on peut bouger.
143 XDino_Draw(vbufID_circle, XDino_TEXID_FONT, g_circlePos);
144
145 // Nombre de millisecondes qu'il a fallu pour afficher la frame précédente.
146 {
147 std::string text = std::format("dTime={:04.1f}ms", deltaTime * 1000.0);
148 std::vector<DinoVertex> vs;
149 Dino_GenVertices_Text(vs, text, DinoColor_WHITE, DinoColor_GREY);
150 uint64_t vbufID = XDino_CreateVertexBuffer(vs.data(), vs.size(), "dTime");
151 XDino_Draw(vbufID, XDino_TEXID_FONT, {}, 2);
152 XDino_DestroyVertexBuffer(vbufID);
153 }
154
155#if !XDINO_RELEASE
156 // Affichage des statistiques si on appuie sur SHIFT.
157 DinoGamepad keyboard;
158 bool bKeyboardOk = XDino_GetGamepad(DinoGamepadIdx::Keyboard, keyboard);
159 if (bKeyboardOk && keyboard.shoulder_left) {
160 int diff = 0;
161 if (keyboard.dpad_up)
162 diff -= 1;
163 if (keyboard.dpad_down)
164 diff += 1;
165 XDino_DrawStats(diff);
166 }
167#endif
168}
