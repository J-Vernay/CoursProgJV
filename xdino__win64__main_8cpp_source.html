41int APIENTRY WinMain(HINSTANCE hInst, HINSTANCE hInstPrev, PSTR cmdline, int cmdshow)
42{
43 // COMMENTAIRE
44 gXDino_hInstance = hInst;
45 XDino_Win64_CreateWindow();
46 ShowWindow(gXDino_hWindow, SW_SHOWNORMAL);
47 UpdateWindow(gXDino_hWindow);
48
49 // COMMENTAIRE
50 MSG msg;
51 BOOL fGotMessage;
52 while ((fGotMessage = GetMessageW(&msg, nullptr, 0, 0))) {
53 if (fGotMessage == -1)
54 DINO_CRITICAL("GetMessageW failed");
55 TranslateMessage(&msg);
56 DispatchMessageW(&msg);
57 }
58
59 // COMMENTAIRE
60 Dino_GameShut();
61 XDino_Win64_DestroyRenderer();
62 XDino_Win64_DestroyWindow();
63
64 return EXIT_SUCCESS;
65}

70void XDino_Win64_CreateWindow()
71{
72 WNDCLASSEXW wcx{};
73 wcx.cbSize = sizeof(WNDCLASSEXW);
74 wcx.lpfnWndProc = &XDino_Win64_HandleEvent;
75 wcx.hInstance = gXDino_hInstance;
76 wcx.style = CS_OWNDC;
77 wcx.lpszClassName = L"Dino";
78
79 ATOM classWindow = RegisterClassExW(&wcx);
80 if (classWindow == 0)
81 DINO_CRITICAL("RegisterClassExW failed");
82
83 RECT windowRect{0, 0, XDino_INIT_WIDTH, XDino_INIT_HEIGHT};
84 AdjustWindowRect(&windowRect, WS_OVERLAPPEDWINDOW, FALSE);
85 int32_t width = windowRect.right - windowRect.left;
86 int32_t height = windowRect.bottom - windowRect.top;
87
88 CreateWindowExW(
89 0,
90 L"Dino",
91 L"Dino",
92 WS_OVERLAPPEDWINDOW,
93 CW_USEDEFAULT,
94 CW_USEDEFAULT,
95 width,
96 height,
97 nullptr,
98 nullptr,
99 gXDino_hInstance,
100 nullptr
101 );
102 // gXDino_hWindow is set by XDino_Win64_HandleEvent, called immediately by CreateWindowExW.
103 if (gXDino_hWindow == nullptr)
104 DINO_CRITICAL("CreateWindowEx failed");
105}

108LRESULT CALLBACK XDino_Win64_HandleEvent(HWND hWnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
109{
110 if (uMsg == WM_CREATE) {
111 // COMMENTAIRE
112 gXDino_hWindow = hWnd;
113
114 LARGE_INTEGER freq;
115 QueryPerformanceFrequency(&freq);
116 gXDino_tickPeriod = static_cast<double>(freq.QuadPart);
117
118 LARGE_INTEGER curTime;
119 QueryPerformanceCounter(&curTime);
120 gXDino_tickStart = curTime.QuadPart;
121
122 RECT windowRect;
123 GetClientRect(gXDino_hWindow, &windowRect);
124 int32_t width = windowRect.right - windowRect.left;
125 int32_t height = windowRect.bottom - windowRect.top;
126 XDino_Win64_CreateRenderer(gXDino_hWindow, width, height);
127
128 Dino_GameInit();
129
130 return 0;
131 }
132 if (uMsg == WM_DESTROY) {
133 // COMMENTAIRE
134 PostQuitMessage(0);
135 return 0;
136 }
137 if (uMsg == WM_SIZE) {
138 // COMMENTAIRE
139 XDino_Win64_ResizeRenderer(LOWORD(lParam), HIWORD(lParam));
140 return 0;
141 }
142 if (uMsg == WM_PAINT) {
143 // COMMENTAIRE
144 XDino_ProfileBegin(DinoColor_BLACK, "Frame");
145 LARGE_INTEGER curTime;
146 QueryPerformanceCounter(&curTime);
147 int64_t tickCount = curTime.QuadPart - gXDino_tickStart;
148 double timeSinceStart = static_cast<double>(tickCount) / gXDino_tickPeriod;
149 XDino_Win64_BeginDraw();
150 Dino_GameFrame(timeSinceStart);
151 XDino_Win64_EndDraw();
152 XDino_ProfileEnd();
153 return 0;
154 }
155 // COMMENTAIRE
156 return DefWindowProcW(hWnd, uMsg, wParam, lParam);
157}

189bool XDino_GetGamepad(DinoGamepadIdx idx, DinoGamepad& outGamepad)
190{
191 outGamepad = {};
192 if (idx == DinoGamepadIdx::Keyboard) {
193 outGamepad.dpad_up = GetAsyncKeyState(VK_UP);
194 outGamepad.dpad_left = GetAsyncKeyState(VK_LEFT);
195 outGamepad.dpad_right = GetAsyncKeyState(VK_RIGHT);
196 outGamepad.dpad_down = GetAsyncKeyState(VK_DOWN);
197 outGamepad.btn_up = GetAsyncKeyState('Z') || GetAsyncKeyState('W');
198 outGamepad.btn_left = GetAsyncKeyState('Q') || GetAsyncKeyState('A');
199 outGamepad.btn_right = GetAsyncKeyState('D');
200 outGamepad.btn_down = GetAsyncKeyState('S');
201 outGamepad.start = GetAsyncKeyState(VK_RETURN) || GetAsyncKeyState(VK_SPACE);
202 outGamepad.select = GetAsyncKeyState(VK_SHIFT);
203 outGamepad.shoulder_left = GetAsyncKeyState(VK_CONTROL);
204 outGamepad.shoulder_right = GetAsyncKeyState(VK_MENU);
205
206 float x, y, hypot;
207
208 x = static_cast<float>(outGamepad.dpad_right) - static_cast<float>(outGamepad.dpad_left);
209 y = static_cast<float>(outGamepad.dpad_down) - static_cast<float>(outGamepad.dpad_up);
210 hypot = std::hypot(x, y);
211 if (hypot > 0) {
212 outGamepad.stick_left_x = x / hypot;
213 outGamepad.stick_left_y = y / hypot;
214 }
215
216 x = static_cast<float>(outGamepad.btn_right) - static_cast<float>(outGamepad.btn_left);
217 y = static_cast<float>(outGamepad.btn_down) - static_cast<float>(outGamepad.btn_up);
218 hypot = std::hypot(x, y);
219 if (hypot > 0) {
220 outGamepad.stick_right_x = x / hypot;
221 outGamepad.stick_right_y = y / hypot;
222 }
223
224 POINT p;
225 if (GetCursorPos(&p) && ScreenToClient(gXDino_hWindow, &p)) {
226 outGamepad.mouse_x = p.x;
227 outGamepad.mouse_y = p.y;
228 }
229
230 return true;
231 }
232
233 int32_t idxNumber = static_cast<int32_t>(idx);
234 if (idxNumber < 0 || idxNumber >= XUSER_MAX_COUNT)
235 return false;
236
237 XINPUT_STATE state;
238 if (XInputGetState(idxNumber, &state) != ERROR_SUCCESS)
239 return false;
240 outGamepad.dpad_up = state.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_UP;
241 outGamepad.dpad_left = state.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_LEFT;
242 outGamepad.dpad_right = state.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_RIGHT;
243 outGamepad.dpad_down = state.Gamepad.wButtons & XINPUT_GAMEPAD_DPAD_DOWN;
244 outGamepad.btn_up = state.Gamepad.wButtons & XINPUT_GAMEPAD_Y;
245 outGamepad.btn_left = state.Gamepad.wButtons & XINPUT_GAMEPAD_X;
246 outGamepad.btn_right = state.Gamepad.wButtons & XINPUT_GAMEPAD_B;
247 outGamepad.btn_down = state.Gamepad.wButtons & XINPUT_GAMEPAD_A;
248 outGamepad.start = state.Gamepad.wButtons & XINPUT_GAMEPAD_START;
249 outGamepad.select = state.Gamepad.wButtons & XINPUT_GAMEPAD_BACK;
250 outGamepad.shoulder_left = state.Gamepad.wButtons & XINPUT_GAMEPAD_LEFT_SHOULDER;
251 outGamepad.shoulder_right = state.Gamepad.wButtons & XINPUT_GAMEPAD_RIGHT_SHOULDER;
252
253 SHORT x, y;
254 x = state.Gamepad.sThumbLX;
255 y = state.Gamepad.sThumbLY;
256 if (x <= -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE || x >= XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE)
257 outGamepad.stick_left_x = static_cast<float>(x) / 32768.f;
258 if (y <= -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE || y >= XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE)
259 outGamepad.stick_left_y = static_cast<float>(y) / -32768.f;
260 x = state.Gamepad.sThumbRX;
261 y = state.Gamepad.sThumbRY;
262 if (x <= -XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE || x >= XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE)
263 outGamepad.stick_right_x = static_cast<float>(x) / 32768.f;
264 if (y <= -XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE || y >= XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE)
265 outGamepad.stick_right_y = static_cast<float>(y) / -32768.f;
266
267 return true;
268}

288DinoVec2 XDino_RandomUnitVec2()
289{
290 // On choisit un angle entre 0 et 2*PI radians.
291 std::uniform_real_distribution<float> distribution(0, 6.28318530718f);
292 float angle = distribution(gXDino_rng);
293 return {cosf(angle), sinf(angle)};
294}
