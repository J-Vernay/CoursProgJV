82constexpr char XDino_VERTEX_SHADER[] = R"(
83cbuffer cb
84{
85 float2 half_vp_size;
86 float2 tex_size;
87 float2 offset;
88 float rot_cos;
89 float rot_sin;
90 float scale;
91}
92
93struct VSInput
94{
95 float2 pos: POSITION0;
96 uint2 uv: TEXCOORD0;
97 float4 color: COLOR0;
98};
99
100struct VSOutput
101{
102 float4 pos: SV_Position;
103 float2 uv: TEXCOORD0;
104 float4 color: COLOR0;
105};
106
107VSOutput main(VSInput input)
108{
109 VSOutput output = (VSOutput)0;
110
111 float2 pos;
112 pos.x = input.pos.x * rot_cos - input.pos.y * rot_sin;
113 pos.y = input.pos.x * rot_sin + input.pos.y * rot_cos;
114 pos *= scale;
115 pos += offset;
116 pos /= half_vp_size;
117 pos -= float2(1, 1);
118
119 // Invert y-axis to have topleft = (0,0)
120 output.pos = float4(pos.x, -pos.y, 0, 1);
121 output.uv = input.uv / tex_size;
122 output.color = input.color;
123 return output;
124}
125)";

156void XDino_Win64_CreateRenderer(HWND hWindow, int32_t width, int32_t height)
157{
158 HRESULT hr = 0;
159
160 gXDino_width = static_cast<float>(width);
161 gXDino_height = static_cast<float>(height);
162
163 DXGI_SWAP_CHAIN_DESC swapchainDesc{};
164 swapchainDesc.BufferDesc.Width = width;
165 swapchainDesc.BufferDesc.Height = height;
166 swapchainDesc.BufferDesc.RefreshRate.Numerator = 1;
167 swapchainDesc.BufferDesc.RefreshRate.Denominator = 60; // 60 fps
168 swapchainDesc.BufferDesc.Format = DXGI_FORMAT_B8G8R8A8_UNORM;
169 swapchainDesc.BufferDesc.ScanlineOrdering = DXGI_MODE_SCANLINE_ORDER_UNSPECIFIED;
170 swapchainDesc.BufferDesc.Scaling = DXGI_MODE_SCALING_STRETCHED;
171 swapchainDesc.SampleDesc.Count = 1;
172 swapchainDesc.SampleDesc.Quality = 0;
173 swapchainDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
174 swapchainDesc.BufferCount = 2;
175 swapchainDesc.OutputWindow = hWindow;
176 swapchainDesc.Windowed = TRUE;
177 swapchainDesc.SwapEffect = DXGI_SWAP_EFFECT_FLIP_DISCARD;
178 swapchainDesc.Flags = 0;
179
180 D3D_FEATURE_LEVEL featureLevel = D3D_FEATURE_LEVEL_11_0;
181 D3D_FEATURE_LEVEL outFeatureLevel = D3D_FEATURE_LEVEL_11_0;
182 hr = D3D11CreateDeviceAndSwapChain(
183 nullptr,
184 D3D_DRIVER_TYPE_HARDWARE,
185 nullptr,
186 D3D11_CREATE_DEVICE_DEBUG,
187 &featureLevel,
188 1,
189 D3D11_SDK_VERSION,
190 &swapchainDesc,
191 &gXDino_swapchain,
192 &gXDino_device,
193 &outFeatureLevel,
194 &gXDino_context
195 );

197 DINO_CRITICAL("D3D11CreateDeviceAndSwapChain failed");
198
199 ID3D11Texture2D* pBackbuffer = nullptr;
200 hr = gXDino_swapchain->GetBuffer(0, IID_ID3D11Texture2D, (void**)&pBackbuffer);
201 if (FAILED(hr))
202 DINO_CRITICAL("gXDino_swapchain->GetBuffer failed");

204 hr = gXDino_device->CreateRenderTargetView(pBackbuffer, nullptr, &gXDino_rtv);
205 if (FAILED(hr))
206 DINO_CRITICAL("gXDino_device->CreateRenderTargetView failed");

210 ID3DBlob* pPixelShaderBlob = nullptr;
211 ID3DBlob* pErrorMsg = nullptr;
212
213 hr = D3DCompile(XDino_VERTEX_SHADER,
214 strlen(XDino_VERTEX_SHADER),
215 nullptr,
216 nullptr,
217 nullptr,
218 "main",

239 DINO_CRITICAL("D3DCompile pixel failed");
240
241 hr = gXDino_device->CreateVertexShader(
242 pVertexShaderBlob->GetBufferPointer(),
243 pVertexShaderBlob->GetBufferSize(),

247 DINO_CRITICAL("CreateVertexShader failed");
248
249 hr = gXDino_device->CreatePixelShader(
250 pPixelShaderBlob->GetBufferPointer(),

358void XDino_Win64_CreateTexture(std::string const& textureName)
359{
360 if (gXDino_textures.contains(textureName))
361 return;
362
363 unsigned char WHITE_TEXTURE[] = {255, 255, 255, 255};
364 HRESULT hr;
365
366 std::string filePath = "assets/" + textureName;
367
368 int x, y, n;
369 unsigned char* pData = stbi_load(filePath.c_str(), &x, &y, &n, 4);
370 if (pData == nullptr) {
371 x = 1;
372 y = 1;
373 pData = WHITE_TEXTURE;
374 }
375
376 XDino_Win64_Texture texture{};
377
378 D3D11_TEXTURE2D_DESC texDesc = {};
379 texDesc.Width = x;
380 texDesc.Height = y;
381 texDesc.MipLevels = 1;
382 texDesc.ArraySize = 1;
383 texDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
384 texDesc.SampleDesc.Count = 1;
385 texDesc.SampleDesc.Quality = 0;
386 texDesc.Usage = D3D11_USAGE_IMMUTABLE;
387 texDesc.BindFlags = D3D11_BIND_SHADER_RESOURCE;
388
389 D3D11_SUBRESOURCE_DATA initData = {};
390 initData.pSysMem = pData;
391 initData.SysMemPitch = 4 * texDesc.Width;
392 initData.SysMemSlicePitch = 4 * texDesc.Width * texDesc.Height;
393 hr = gXDino_device->CreateTexture2D(&texDesc, &initData, &texture.pTexture);
394 if (FAILED(hr))
395 DINO_CRITICAL("CreateTexture2D failed");
396
397 D3D11_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
398 srvDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
399 srvDesc.ViewDimension = D3D11_SRV_DIMENSION_TEXTURE2D;
400 srvDesc.Texture2D.MostDetailedMip = 0;
401 srvDesc.Texture2D.MipLevels = 1;
402 hr = gXDino_device->CreateShaderResourceView(texture.pTexture, &srvDesc, &texture.pTextureView);
403 if (FAILED(hr))
404 DINO_CRITICAL("CreateShaderResourceView failed");
405
406 D3D11_SAMPLER_DESC samplerDesc = {};
407 samplerDesc.Filter = D3D11_FILTER_MIN_MAG_POINT_MIP_LINEAR;
408 samplerDesc.AddressU = D3D11_TEXTURE_ADDRESS_CLAMP;
409 samplerDesc.AddressV = D3D11_TEXTURE_ADDRESS_CLAMP;
410 samplerDesc.AddressW = D3D11_TEXTURE_ADDRESS_CLAMP;
411 hr = gXDino_device->CreateSamplerState(&samplerDesc, &texture.pTextureSampler);
412 if (FAILED(hr))
413 DINO_CRITICAL("CreateSamplerState failed");
414
415 if (pData != WHITE_TEXTURE) {
416 stbi_image_free(pData);
417 }
418
419 texture.texWidth = static_cast<uint16_t>(x);
420 texture.texHeight = static_cast<uint16_t>(y);
421 gXDino_textures.emplace(textureName, texture);
422}

429void XDino_Win64_EndDraw()
430{
431 XDino_ProfileBegin(DinoColor_GREY, "XDino_Win64_EndDraw");
432
433 DinoColor color = gXDino_clearColor;
434 float clearColor[4] = {color.r / 255.f, color.g / 255.f, color.b / 255.f, color.a / 255.f};
435 gXDino_context->ClearRenderTargetView(gXDino_rtv, clearColor);
436
437 UINT stride = sizeof(DinoVertex);
438 UINT offset = 0;
439 gXDino_context->IASetInputLayout(gXDino_inputLayout);
440 gXDino_context->IASetVertexBuffers(0, 1, &gXDino_vertexBuffer, &stride, &offset);
441 gXDino_context->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
442
443 gXDino_context->VSSetShader(gXDino_vertexShader, nullptr, 0);
444 gXDino_context->VSSetConstantBuffers(0, 1, &gXDino_constantBuffer);
445
446 float width = gXDino_width;
447 float height = gXDino_rdrHeight * gXDino_width / gXDino_rdrWidth;
448 if (height > gXDino_height) {
449 width = gXDino_rdrWidth * gXDino_height / gXDino_rdrHeight;
450 height = gXDino_height;
451 }
452
453 D3D11_VIEWPORT viewport = {};
454 viewport.TopLeftX = (gXDino_width - width) / 2;
455 viewport.TopLeftY = (gXDino_height - height) / 2;
456 viewport.Width = width;
457 viewport.Height = height;
458 viewport.MinDepth = 0.0f;
459 viewport.MaxDepth = 1.0f;
460 gXDino_context->RSSetViewports(1, &viewport);
461 gXDino_context->RSSetState(gXDino_rasterState);
462
463 gXDino_context->PSSetShader(gXDino_pixelShader, nullptr, 0);
464
465 gXDino_context->OMSetRenderTargets(1, &gXDino_rtv, nullptr);
466
467 gXDino_context->OMSetBlendState(gXDino_blendState, nullptr, 0xFFFFFFFF);
468
469 for (DinoDrawCall const& drawCall : gXDino_drawCalls) {
470 D3D11_MAPPED_SUBRESOURCE resource;
471
472 // Envoyer la texture à la carte graphique, si elle n'existe pas déjà.
473
474 if (!gXDino_textures.contains(drawCall.textureName))
475 XDino_Win64_CreateTexture(drawCall.textureName);
476 XDino_Win64_Texture const& texture = gXDino_textures[drawCall.textureName];
477 if (texture.pTexture == nullptr)
478 DINO_CRITICAL("XDino_Win64_CreateTexture failed");
479
480 gXDino_context->PSSetShaderResources(0, 1, &texture.pTextureView);
481 gXDino_context->PSSetSamplers(0, 1, &texture.pTextureSampler);
482
483 // Mettre à jour les transformations.
484
485 double rotationRadians = drawCall.rotation * (std::numbers::pi / 180.0);
486 XDino_Win64_CBuffer cbuffer{};
487 cbuffer.half_vp_size_x = gXDino_rdrWidth / 2.f;
488 cbuffer.half_vp_size_y = gXDino_rdrHeight / 2.f;
489 cbuffer.tex_size_x = static_cast<float>(texture.texWidth);
490 cbuffer.tex_size_y = static_cast<float>(texture.texHeight);
491 cbuffer.offset_x = static_cast<float>(drawCall.translation.x);
492 cbuffer.offset_y = static_cast<float>(drawCall.translation.y);
493 cbuffer.rot_cos = static_cast<float>(std::cos(rotationRadians));
494 cbuffer.rot_sin = static_cast<float>(std::sin(rotationRadians));
495 cbuffer.scale = static_cast<float>(drawCall.scale);
496
497 gXDino_context->Map(gXDino_constantBuffer, 0, D3D11_MAP_WRITE_DISCARD, 0, &resource);
498 memcpy(resource.pData, &cbuffer, sizeof(cbuffer));
499 gXDino_context->Unmap(gXDino_constantBuffer, 0);
500
501 // Envoyer les sommets de triangle à la carte graphique.
502
503 XDino_ProfileBegin({0x44, 0x44, 0x44, 0xFF},
504 std::format("Map {} vertices ({})", drawCall.vertices.size(), drawCall.textureName).c_str());
505 gXDino_context->Map(gXDino_vertexBuffer, 0, D3D11_MAP_WRITE_DISCARD, 0, &resource);
506 memcpy(resource.pData, drawCall.vertices.data(), drawCall.vertices.size() * sizeof(DinoVertex));
507 gXDino_context->Unmap(gXDino_vertexBuffer, 0);
508 XDino_ProfileEnd();
509
510 XDino_ProfileBegin({0x44, 0x44, 0x44, 0xFF}, std::format("Draw {}", drawCall.textureName).c_str());
511 gXDino_context->Draw(static_cast<UINT>(drawCall.vertices.size()), 0);
512 XDino_ProfileEnd();
513 }
514
515 XDino_ProfileEnd();
516
517 XDino_ProfileBegin({0x44, 0x44, 0x44, 0xFF}, "SwapchainPresent");
518 gXDino_swapchain->Present(1, 0);
519 XDino_ProfileEnd();
520}
