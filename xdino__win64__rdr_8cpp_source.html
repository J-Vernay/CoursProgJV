93std::vector<std::string> XDino_Win64_CollectRessources(bool bIncludeTemp = false)
94{
95 std::vector<std::string> lines;
96 lines.emplace_back(std::format("[{} Textures]", gXDino_textures.size()));
97 for (auto& [id, tex] : gXDino_textures) {
98 if (bIncludeTemp || !tex.bDestroy)
99 lines.push_back(
100 std::format("{}. {} ({}x{}, {} bytes)", id, tex.name, tex.width, tex.height, tex.width * tex.height * 4)
101 );
102 }
103 lines.emplace_back();
104 lines.emplace_back(std::format("[{} VertexBuffers]", gXDino_vertexBuffers.size()));
105 for (auto& [id, vbuf] : gXDino_vertexBuffers) {
106 if (bIncludeTemp || !vbuf.bDestroy)
107 lines.push_back(std::format("{}. {} ({} vertices)", id, vbuf.name, vbuf.count));
108 }
109 lines.emplace_back();
110 lines.emplace_back(std::format("[{} MemAllocs]", gXDino_allocs.size()));
111 for (auto& [id, alloc] : gXDino_allocs) {
112 if (bIncludeTemp || !alloc.bDestroy)
113 lines.push_back(std::format("{}. {} ({} bytes)", id, alloc.name, alloc.size));
114 }
115 return lines;
116}

118void XDino_Win64_PurgeDeadResources()
119{
120 for (auto it = gXDino_textures.begin(); it != gXDino_textures.end();) {
121 if (it->second.bDestroy)
122 it = gXDino_textures.erase(it);
123 else
124 ++it;
125 }
126 for (auto it = gXDino_vertexBuffers.begin(); it != gXDino_vertexBuffers.end();) {
127 if (it->second.bDestroy)
128 it = gXDino_vertexBuffers.erase(it);
129 else
130 ++it;
131 }
132 for (auto it = gXDino_allocs.begin(); it != gXDino_allocs.end();) {
133 if (it->second.bDestroy)
134 it = gXDino_allocs.erase(it);
135 else
136 ++it;
137 }
138}

144struct alignas(16) XDino_Win64_CBuffer {
145 float half_vp_size_x;
146 float half_vp_size_y;
147 float tex_size_x;
148 float tex_size_y;
149 float offset_x;
150 float offset_y;
151 float rot_cos;
152 float rot_sin;
153 float scale;
154};
155
156constexpr char XDino_VERTEX_SHADER[] = R"(
157cbuffer cb
158{
159 float2 half_vp_size;
160 float2 tex_size;
161 float2 offset;
162 float rot_cos;
163 float rot_sin;
164 float scale;
165}
166
167struct VSInput
168{
169 float2 pos: POSITION0;
170 uint2 uv: TEXCOORD0;
171 float4 color: COLOR0;
172};
173
174struct VSOutput
175{
176 float4 pos: SV_Position;
177 float2 uv: TEXCOORD0;
178 float4 color: COLOR0;
179};
180
181VSOutput main(VSInput input)
182{
183 VSOutput output = (VSOutput)0;
184
185 float2 pos;
186 pos.x = input.pos.x * rot_cos - input.pos.y * rot_sin;
187 pos.y = input.pos.x * rot_sin + input.pos.y * rot_cos;
188 pos *= scale;
189 pos += offset;
190 pos /= half_vp_size;
191 pos -= float2(1, 1);
192
193 // Invert y-axis to have topleft = (0,0)
194 output.pos = float4(pos.x, -pos.y, 0, 1);
195 output.uv = input.uv / tex_size;
196 output.color = input.color;
197 return output;
198}
199)";
200
201constexpr char XDino_PIXEL_SHADER[] = R"(
202sampler Sampler : register(s0);
203Texture2D Texture : register(t0);
204
205struct PSInput
206{
207 float4 pos: SV_Position;
208 float2 uv: TEXCOORD0;
209 float4 color: COLOR0;
210};
211
212struct PSOutput
213{
214 float4 color: SV_Target0;
215};
216
217PSOutput main(PSInput input)
218{
219 PSOutput output = (PSOutput)0;
220 output.color = Texture.Sample(Sampler, input.uv);
221 output.color *= input.color;
222 return output;
223}
224)";

230void XDino_Win64_CreateRenderer(HWND hWindow, int32_t width, int32_t height)
231{
232 HRESULT hr = 0;
233
234 gXDino_width = static_cast<float>(width);
235 gXDino_height = static_cast<float>(height);
236
237 DXGI_SWAP_CHAIN_DESC swapchainDesc{};
238 swapchainDesc.BufferDesc.Width = width;
239 swapchainDesc.BufferDesc.Height = height;
240 swapchainDesc.BufferDesc.RefreshRate.Numerator = 1;
241 swapchainDesc.BufferDesc.RefreshRate.Denominator = 60; // 60 fps
242 swapchainDesc.BufferDesc.Format = DXGI_FORMAT_B8G8R8A8_UNORM;
243 swapchainDesc.BufferDesc.ScanlineOrdering = DXGI_MODE_SCANLINE_ORDER_UNSPECIFIED;
244 swapchainDesc.BufferDesc.Scaling = DXGI_MODE_SCALING_STRETCHED;
245 swapchainDesc.SampleDesc.Count = 1;
246 swapchainDesc.SampleDesc.Quality = 0;
247 swapchainDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
248 swapchainDesc.BufferCount = 2;
249 swapchainDesc.OutputWindow = hWindow;
250 swapchainDesc.Windowed = TRUE;
251 swapchainDesc.SwapEffect = DXGI_SWAP_EFFECT_FLIP_DISCARD;
252 swapchainDesc.Flags = 0;
253
254 D3D_FEATURE_LEVEL featureLevel = D3D_FEATURE_LEVEL_11_0;
255 D3D_FEATURE_LEVEL outFeatureLevel = D3D_FEATURE_LEVEL_11_0;
256 hr = D3D11CreateDeviceAndSwapChain(
257 nullptr,
258 D3D_DRIVER_TYPE_HARDWARE,
259 nullptr,
260 0,
261 //D3D11_CREATE_DEVICE_DEBUG,
262 &featureLevel,
263 1,
264 D3D11_SDK_VERSION,
265 &swapchainDesc,
266 &gXDino_swapchain,
267 &gXDino_device,
268 &outFeatureLevel,
269 &gXDino_context
270 );
271 if (FAILED(hr))
272 DINO_CRITICAL("D3D11CreateDeviceAndSwapChain failed");
273
274 ID3D11Texture2D* pBackbuffer = nullptr;
275 hr = gXDino_swapchain->GetBuffer(0, IID_ID3D11Texture2D, (void**)&pBackbuffer);
276 if (FAILED(hr))
277 DINO_CRITICAL("gXDino_swapchain->GetBuffer failed");
278
279 hr = gXDino_device->CreateRenderTargetView(pBackbuffer, nullptr, &gXDino_rtv);
280 if (FAILED(hr))
281 DINO_CRITICAL("gXDino_device->CreateRenderTargetView failed");
282 pBackbuffer->Release();
283
284 ID3DBlob* pVertexShaderBlob = nullptr;
285 ID3DBlob* pPixelShaderBlob = nullptr;
286 ID3DBlob* pErrorMsg = nullptr;
287
288 hr = D3DCompile(
289 XDino_VERTEX_SHADER,
290 strlen(XDino_VERTEX_SHADER),
291 nullptr,
292 nullptr,
293 nullptr,
294 "main",
295 "vs_5_0",
296 0,
297 0,
298 &pVertexShaderBlob,
299 &pErrorMsg
300 );
301 if (FAILED(hr))
302 DINO_CRITICAL("D3DCompile vertex failed");
303
304 hr = D3DCompile(
305 XDino_PIXEL_SHADER,
306 strlen(XDino_PIXEL_SHADER),
307 nullptr,
308 nullptr,
309 nullptr,
310 "main",
311 "ps_5_0",
312 0,
313 0,
314 &pPixelShaderBlob,
315 &pErrorMsg
316 );
317 if (FAILED(hr))
318 DINO_CRITICAL("D3DCompile pixel failed");
319
320 hr = gXDino_device->CreateVertexShader(
321 pVertexShaderBlob->GetBufferPointer(),
322 pVertexShaderBlob->GetBufferSize(),
323 nullptr,
324 &gXDino_vertexShader
325 );
326 if (FAILED(hr))
327 DINO_CRITICAL("CreateVertexShader failed");
328
329 hr = gXDino_device->CreatePixelShader(
330 pPixelShaderBlob->GetBufferPointer(),
331 pPixelShaderBlob->GetBufferSize(),
332 nullptr,
333 &gXDino_pixelShader
334 );
335 if (FAILED(hr))
336 DINO_CRITICAL("CreatePixelShader failed");
337
338 D3D11_INPUT_ELEMENT_DESC inputs[3]{};
339 inputs[0].SemanticName = "POSITION";
340 inputs[0].SemanticIndex = 0;
341 inputs[0].Format = DXGI_FORMAT_R32G32_FLOAT;
342 inputs[0].InputSlot = 0;
343 inputs[0].AlignedByteOffset = offsetof(DinoVertex, pos);
344 inputs[0].InputSlotClass = D3D11_INPUT_PER_VERTEX_DATA;
345 inputs[1].SemanticName = "TEXCOORD";
346 inputs[1].SemanticIndex = 0;
347 inputs[1].Format = DXGI_FORMAT_R16G16_UINT;
348 inputs[1].InputSlot = 0;
349 inputs[1].AlignedByteOffset = offsetof(DinoVertex, u);
350 inputs[1].InputSlotClass = D3D11_INPUT_PER_VERTEX_DATA;
351 inputs[2].SemanticName = "COLOR";
352 inputs[2].SemanticIndex = 0;
353 inputs[2].Format = DXGI_FORMAT_R8G8B8A8_UNORM;
354 inputs[2].InputSlot = 0;
355 inputs[2].AlignedByteOffset = offsetof(DinoVertex, color);
356 inputs[2].InputSlotClass = D3D11_INPUT_PER_VERTEX_DATA;
357
358 hr = gXDino_device->CreateInputLayout(
359 inputs,
360 3,
361 pVertexShaderBlob->GetBufferPointer(),
362 pVertexShaderBlob->GetBufferSize(),
363 &gXDino_inputLayout
364 );
365 if (FAILED(hr))
366 DINO_CRITICAL("CreateInputLayout failed");
367
368 pVertexShaderBlob->Release();
369 pPixelShaderBlob->Release();
370
371 D3D11_BUFFER_DESC cbufferDesc = {};
372 cbufferDesc.ByteWidth = sizeof(XDino_Win64_CBuffer);
373 cbufferDesc.Usage = D3D11_USAGE_DYNAMIC;
374 cbufferDesc.BindFlags = D3D11_BIND_CONSTANT_BUFFER;
375 cbufferDesc.CPUAccessFlags = D3D11_CPU_ACCESS_WRITE;
376
377 hr = gXDino_device->CreateBuffer(&cbufferDesc, nullptr, &gXDino_constantBuffer);
378 if (FAILED(hr))
379 DINO_CRITICAL("CreateBuffer constant failed");
380
381 D3D11_RASTERIZER_DESC rasterDesc = {};
382 rasterDesc.FillMode = D3D11_FILL_SOLID;
383 rasterDesc.CullMode = D3D11_CULL_NONE;
384
385 hr = gXDino_device->CreateRasterizerState(&rasterDesc, &gXDino_rasterState);
386 if (FAILED(hr))
387 DINO_CRITICAL("CreateRasterizerState failed");
388
389 D3D11_BLEND_DESC blendDesc = {0};
390 blendDesc.RenderTarget[0].BlendEnable = TRUE;
391 blendDesc.RenderTarget[0].SrcBlend = D3D11_BLEND_SRC_ALPHA;
392 blendDesc.RenderTarget[0].DestBlend = D3D11_BLEND_INV_SRC_ALPHA;
393 blendDesc.RenderTarget[0].BlendOp = D3D11_BLEND_OP_ADD;
394 blendDesc.RenderTarget[0].SrcBlendAlpha = D3D11_BLEND_ZERO;
395 blendDesc.RenderTarget[0].DestBlendAlpha = D3D11_BLEND_ONE;
396 blendDesc.RenderTarget[0].BlendOpAlpha = D3D11_BLEND_OP_ADD;
397 blendDesc.RenderTarget[0].RenderTargetWriteMask =
398 D3D11_COLOR_WRITE_ENABLE_RED | D3D11_COLOR_WRITE_ENABLE_GREEN | D3D11_COLOR_WRITE_ENABLE_BLUE;
399 hr = gXDino_device->CreateBlendState(&blendDesc, &gXDino_blendState);
400 if (FAILED(hr))
401 DINO_CRITICAL("CreateBlendState failed");
402
403 unsigned char WHITE_TEXTURE[] = {255, 255, 255, 255};
404 uint64_t texID = XDino_Win64_CreateTexture("WHITE", 1, 1, WHITE_TEXTURE);
405 if (texID != XDino_TEXID_WHITE)
406 DINO_CRITICAL("TEXID_WHITE illogique.");
407 texID = XDino_CreateGpuTexture("monogram-bitmap.png");
408 if (texID != XDino_TEXID_FONT)
409 DINO_CRITICAL("TEXID_FONT illogique.");
410
411 gXDino_vertexBuffers.emplace(0, "EMPTY");
412}

438uint64_t XDino_Win64_CreateTexture(char const* pName, int w, int h, void const* pData)
439{
440 HRESULT hr;
441
442 XDino_Win64_Texture texture{};
443
444 D3D11_TEXTURE2D_DESC texDesc = {};
445 texDesc.Width = w;
446 texDesc.Height = h;
447 texDesc.MipLevels = 1;
448 texDesc.ArraySize = 1;
449 texDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
450 texDesc.SampleDesc.Count = 1;
451 texDesc.SampleDesc.Quality = 0;
452 texDesc.Usage = D3D11_USAGE_IMMUTABLE;
453 texDesc.BindFlags = D3D11_BIND_SHADER_RESOURCE;
454
455 D3D11_SUBRESOURCE_DATA initData = {};
456 initData.pSysMem = pData;
457 initData.SysMemPitch = 4 * texDesc.Width;
458 initData.SysMemSlicePitch = 4 * texDesc.Width * texDesc.Height;
459 hr = gXDino_device->CreateTexture2D(&texDesc, &initData, &texture.pTexture);
460 if (FAILED(hr))
461 DINO_CRITICAL("CreateTexture2D failed");
462
463 D3D11_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
464 srvDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
465 srvDesc.ViewDimension = D3D11_SRV_DIMENSION_TEXTURE2D;
466 srvDesc.Texture2D.MostDetailedMip = 0;
467 srvDesc.Texture2D.MipLevels = 1;
468 hr = gXDino_device->CreateShaderResourceView(texture.pTexture, &srvDesc, &texture.pTextureView);
469 if (FAILED(hr))
470 DINO_CRITICAL("CreateShaderResourceView failed");
471
472 D3D11_SAMPLER_DESC samplerDesc = {};
473 samplerDesc.Filter = D3D11_FILTER_MIN_MAG_POINT_MIP_LINEAR;
474 samplerDesc.AddressU = D3D11_TEXTURE_ADDRESS_CLAMP;
475 samplerDesc.AddressV = D3D11_TEXTURE_ADDRESS_CLAMP;
476 samplerDesc.AddressW = D3D11_TEXTURE_ADDRESS_CLAMP;
477 hr = gXDino_device->CreateSamplerState(&samplerDesc, &texture.pTextureSampler);
478 if (FAILED(hr))
479 DINO_CRITICAL("CreateSamplerState failed");
480
481 uint64_t texID = gXDino_textureCounter++;
482 texture.name = pName;
483 texture.width = static_cast<uint16_t>(w);
484 texture.height = static_cast<uint16_t>(h);
485 gXDino_textures.emplace(texID, texture);
486 return texID;
487}

489void XDino_Win64_BeginDraw()
490{
491 DinoColor color = gXDino_clearColor;
492 float clearColor[4] = {color.r / 255.f, color.g / 255.f, color.b / 255.f, color.a / 255.f};
493 gXDino_context->ClearRenderTargetView(gXDino_rtv, clearColor);
494
495 gXDino_context->IASetInputLayout(gXDino_inputLayout);
496 gXDino_context->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
497
498 gXDino_context->VSSetShader(gXDino_vertexShader, nullptr, 0);
499 gXDino_context->VSSetConstantBuffers(0, 1, &gXDino_constantBuffer);
500
501 float width = gXDino_width;
502 float height = gXDino_rdrHeight * gXDino_width / gXDino_rdrWidth;
503 if (height > gXDino_height) {
504 width = gXDino_rdrWidth * gXDino_height / gXDino_rdrHeight;
505 height = gXDino_height;
506 }
507
508 D3D11_VIEWPORT viewport = {};
509 viewport.TopLeftX = (gXDino_width - width) / 2;
510 viewport.TopLeftY = (gXDino_height - height) / 2;
511 viewport.Width = width;
512 viewport.Height = height;
513 viewport.MinDepth = 0.0f;
514 viewport.MaxDepth = 1.0f;
515 gXDino_context->RSSetViewports(1, &viewport);
516 gXDino_context->RSSetState(gXDino_rasterState);
517
518 gXDino_context->PSSetShader(gXDino_pixelShader, nullptr, 0);
519
520 gXDino_context->OMSetRenderTargets(1, &gXDino_rtv, nullptr);
521
522 gXDino_context->OMSetBlendState(gXDino_blendState, nullptr, 0xFFFFFFFF);
523}

525void XDino_Win64_EndDraw()
526{
527 if (gXDino_bDrawStats) {
528 std::vector<std::string> lines = XDino_Win64_CollectRessources(true);
529
530 gXDino_statScroll += gXDino_statScrollUser;
531 gXDino_statScroll = XDinoImpl_DrawStats(gXDino_statScroll, gXDino_rdrHeight, 1, std::move(lines));
532
533 gXDino_bDrawStats = false;
534 }
535 XDino_Win64_PurgeDeadResources();
536 XDino_ProfileBegin({0x44, 0x44, 0x44, 0xFF}, "SwapchainPresent");
537 gXDino_swapchain->Present(1, 0);
538 XDino_ProfileEnd();
539}

541void XDino_Win64_DestroyRenderer()
542{
543 XDino_DestroyVertexBuffer(XDino_VBUFID_EMPTY);
544 XDino_DestroyGpuTexture(XDino_TEXID_FONT);
545 XDino_DestroyGpuTexture(XDino_TEXID_WHITE);
546 XDino_Win64_PurgeDeadResources();
547
548 OutputDebugStringA("--- RESOURCES ALIVE BEGIN ---\n");
549 for (std::string const& s : XDino_Win64_CollectRessources())
550 OutputDebugStringA((s + "\n").c_str());
551 OutputDebugStringA("--- RESOURCES ALIVE END ---\n");
552 std::puts("--- RESOURCES ALIVE BEGIN ---");
553 for (std::string s : XDino_Win64_CollectRessources())
554 std::puts(s.c_str());
555 std::puts("--- RESOURCES ALIVE END ---");
556
557 for (auto& [name, texture] : gXDino_textures) {
558 texture.pTextureSampler->Release();
559 texture.pTextureView->Release();
560 texture.pTexture->Release();
561 }
562 gXDino_textures.clear();
563
564 gXDino_blendState->Release();
565 gXDino_blendState = nullptr;
566 gXDino_rasterState->Release();
567 gXDino_rasterState = nullptr;
568 gXDino_constantBuffer->Release();
569 gXDino_constantBuffer = nullptr;

571 gXDino_inputLayout = nullptr;
572 gXDino_pixelShader->Release();
573 gXDino_pixelShader = nullptr;
574 gXDino_vertexShader->Release();
575 gXDino_vertexShader = nullptr;
576 gXDino_rtv->Release();
577 gXDino_rtv = nullptr;
578 gXDino_context->Release();
579 gXDino_context = nullptr;
580 gXDino_swapchain->Release();
581 gXDino_swapchain = nullptr;

583 gXDino_device = nullptr;
584}

615uint64_t XDino_CreateGpuTexture(char const* pName)
616{
617 std::string filePath = std::format("assets/{}", pName);
618 int w, h, n;
619 unsigned char* pData = stbi_load(filePath.c_str(), &w, &h, &n, 4);

653{
654 XDino_ProfileBegin(
655 {0x44,
656 0x44,
657 0x44,
658 0xFF},
659 std::format("Create vertex buffer '{}' ({})", vertexCount, pLabel).c_str()
660 );
661
662 ID3D11Buffer* buf = nullptr;
663 if (vertexCount > 0) {
664 D3D11_BUFFER_DESC bufferDesc = {};
665 bufferDesc.ByteWidth = static_cast<UINT>(sizeof(DinoVertex) * vertexCount);
666 bufferDesc.Usage = D3D11_USAGE_IMMUTABLE;
667 bufferDesc.BindFlags = D3D11_BIND_VERTEX_BUFFER;
668
669 D3D11_SUBRESOURCE_DATA bufferData = {};
670 bufferData.pSysMem = pVertices;
671
672 HRESULT hr = gXDino_device->CreateBuffer(&bufferDesc, &bufferData, &buf);
673 if (FAILED(hr))
674 DINO_CRITICAL("CreateBuffer vertex failed");
675 }
676
677 uint64_t vbufID = gXDino_vertexBufferCounter++;
678 XDino_Win64_VertexBuffer vbuf;
679 vbuf.name = pLabel;
680 vbuf.count = vertexCount;
681 vbuf.buffer = buf;
682 gXDino_vertexBuffers.emplace(vbufID, vbuf);
683 XDino_ProfileEnd();
684 return vbufID;
685}

699void XDino_Draw(uint64_t vbufID, uint64_t texID, DinoVec2 translation, double scale, double rotation)
700{
701 D3D11_MAPPED_SUBRESOURCE resource;
702
703 auto itTex = gXDino_textures.find(texID);
704 if (itTex == gXDino_textures.end() || itTex->second.bDestroy)
705 DINO_CRITICAL("Impossible de trouver la texture pour le drawcall");
706 XDino_Win64_Texture const& texture = itTex->second;
707
708 if (vbufID == 0)
709 return;
710
711 auto itVbuf = gXDino_vertexBuffers.find(vbufID);
712 if (itVbuf == gXDino_vertexBuffers.end() || itVbuf->second.bDestroy)
713 DINO_CRITICAL("Impossible de trouver la texture pour le drawcall");
714 XDino_Win64_VertexBuffer const& vbuf = itVbuf->second;
715
716 if (vbuf.count == 0)
717 return;
718
719 XDino_ProfileBegin({0x44, 0x44, 0x44, 0xFF}, std::format("Draw {} {}", texture.name, vbuf.name).c_str());
720
721 gXDino_context->PSSetShaderResources(0, 1, &texture.pTextureView);
722 gXDino_context->PSSetSamplers(0, 1, &texture.pTextureSampler);
723
724 // Mettre à jour les transformations.
725
726 double rotationRadians = rotation * (std::numbers::pi / 180.0);
727 XDino_Win64_CBuffer cbuffer{};
728 cbuffer.half_vp_size_x = gXDino_rdrWidth / 2.f;
729 cbuffer.half_vp_size_y = gXDino_rdrHeight / 2.f;
730 cbuffer.tex_size_x = static_cast<float>(texture.width);
731 cbuffer.tex_size_y = static_cast<float>(texture.height);
732 cbuffer.offset_x = static_cast<float>(translation.x);
733 cbuffer.offset_y = static_cast<float>(translation.y);
734 cbuffer.rot_cos = static_cast<float>(std::cos(rotationRadians));
735 cbuffer.rot_sin = static_cast<float>(std::sin(rotationRadians));
736 cbuffer.scale = static_cast<float>(scale);
737
738 gXDino_context->Map(gXDino_constantBuffer, 0, D3D11_MAP_WRITE_DISCARD, 0, &resource);
739 memcpy(resource.pData, &cbuffer, sizeof(cbuffer));
740 gXDino_context->Unmap(gXDino_constantBuffer, 0);
741
742 UINT stride = sizeof(DinoVertex);
743 UINT offset = 0;
744 gXDino_context->IASetVertexBuffers(0, 1, &vbuf.buffer, &stride, &offset);
745 gXDino_context->Draw(static_cast<UINT>(vbuf.count), 0);
746 XDino_ProfileEnd();
747}

759void* XDino_MemAlloc(size_t size, char const* pLabel)
760{
761 if (size == 0)
762 DINO_CRITICAL("Allouer 0 octets n'a pas de sens.");
763 void* p = VirtualAlloc(nullptr, size, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
764 if (!p)
765 DINO_CRITICAL("L'OS a échoué à allouer de la mémoire.");
766 XDino_Win64_Alloc& a = gXDino_allocs[p];
767 a.size = size;
768 a.name = pLabel;
769 a.bDestroy = false;
770 memset(p, 0xCC, size);
771 return p;
772}

774void XDino_MemFree(void* pAlloc)
775{
776 if (!pAlloc)
777 return;
778 auto it = gXDino_allocs.find(pAlloc);
779 if (it == gXDino_allocs.end() || it->second.size == 0 || it->second.bDestroy)
780 DINO_CRITICAL("Libération de mémoire non-allouée.");
781 if (it->second.size == 0)
782 DINO_CRITICAL("Libération de mémoire qui a déjà été allouée.");
783 VirtualFree(pAlloc, 0, MEM_RELEASE);
784 it->second.bDestroy = true;
785}
