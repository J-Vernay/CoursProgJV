118 }

120 lines.emplace_back(std::format("[{} MemAllocs]", gXDino_allocs.size()));
121 for (auto& [id, alloc] : gXDino_allocs) {
122 if (bIncludeTemp || !alloc.bDestroy)
123 lines.push_back(std::format("{}. {} ({} bytes)", id, alloc.name, alloc.size));
124 }

232{

234 pass.action = gXDino_gfxPassAction;

238

240 float vpw = vpscale * gXDino_renderSize.x;

276void XDino_Linux_Event(sapp_event const* pEvent)
277{
278 if (pEvent->type == SAPP_EVENTTYPE_RESIZED) {
279 gXDino_windowSize = {(float)pEvent->window_width, (float)pEvent->window_height};
280 }
281 else if (pEvent->type == SAPP_EVENTTYPE_KEY_DOWN || pEvent->type == SAPP_EVENTTYPE_KEY_UP) {
282 // Keyboard
283 DinoGamepad& g = gXDino_gamepads[(uint32_t)DinoGamepadIdx::Keyboard].g;
284 bool bPressed = pEvent->type == SAPP_EVENTTYPE_KEY_DOWN;
285 switch (pEvent->key_code) {
286 case SAPP_KEYCODE_UP: g.dpad_up = bPressed; break;
287 case SAPP_KEYCODE_LEFT: g.dpad_left = bPressed; break;
288 case SAPP_KEYCODE_RIGHT: g.dpad_right = bPressed; break;
289 case SAPP_KEYCODE_DOWN: g.dpad_down = bPressed; break;
290 case SAPP_KEYCODE_Z:
291 case SAPP_KEYCODE_W: g.btn_up = bPressed; break; // Z ou W
292 case SAPP_KEYCODE_Q:
293 case SAPP_KEYCODE_A: g.btn_left = bPressed; break; // Q ou A
294 case SAPP_KEYCODE_D: g.btn_right = bPressed; break;
295 case SAPP_KEYCODE_S: g.btn_down = bPressed; break;
296 case SAPP_KEYCODE_SPACE:
297 case SAPP_KEYCODE_ENTER: g.start = bPressed; break;
298 case SAPP_KEYCODE_LEFT_SHIFT:
299 case SAPP_KEYCODE_RIGHT_SHIFT: g.select = bPressed; break;
300 case SAPP_KEYCODE_LEFT_CONTROL:
301 case SAPP_KEYCODE_RIGHT_CONTROL: g.shoulder_left = bPressed; break;
302 case SAPP_KEYCODE_LEFT_ALT:
303 case SAPP_KEYCODE_RIGHT_ALT: g.shoulder_right = bPressed; break;
304 default: break;
305 }
306
307 float x, y, hypot;
308
309 x = static_cast<float>(g.dpad_right) - static_cast<float>(g.dpad_left);
310 y = static_cast<float>(g.dpad_down) - static_cast<float>(g.dpad_up);
311 hypot = std::hypot(x, y);
312 if (hypot > 0) {
313 g.stick_left_x = x / hypot;
314 g.stick_left_y = y / hypot;
315 }
316 else {
317 g.stick_left_x = 0;
318 g.stick_left_y = 0;
319 }
320
321 x = static_cast<float>(g.btn_right) - static_cast<float>(g.btn_left);
322 y = static_cast<float>(g.btn_down) - static_cast<float>(g.btn_up);
323 hypot = std::hypot(x, y);
324 if (hypot > 0) {
325 g.stick_right_x = x / hypot;
326 g.stick_right_y = y / hypot;
327 }
328 else {
329 g.stick_right_x = 0;
330 g.stick_right_y = 0;
331 }
332 }
333 else if (pEvent->type == SAPP_EVENTTYPE_MOUSE_MOVE) {
334 DinoGamepad& g = gXDino_gamepads[(uint32_t)DinoGamepadIdx::Keyboard].g;
335 g.mouse_x = pEvent->mouse_x;
336 g.mouse_y = pEvent->mouse_y;
337 }
338}

454void XDino_Draw(uint64_t vbufID, uint64_t texID, DinoVec2 translation, double scale, double rotation)
455{
456 auto itTex = gXDino_textures.find(texID);
457 if (itTex == gXDino_textures.end() || itTex->second.bDestroy)
458 DINO_CRITICAL("Impossible de trouver la texture pour le drawcall");
459 XDino_Linux_Texture const& tex = itTex->second;
460
461 if (vbufID == 0)
462 return;
463
464 auto itVbuf = gXDino_vertexBuffers.find(vbufID);
465 if (itVbuf == gXDino_vertexBuffers.end() || itVbuf->second.bDestroy)
466 DINO_CRITICAL("Impossible de trouver la texture pour le drawcall");
467 XDino_Linux_VertexBuffer const& vbuf = itVbuf->second;
468
469 if (vbuf.count == 0)
470 return;
471
472 sg_bindings bindings{};
473 bindings.vertex_buffers[0] = vbuf.buffer;
474 bindings.views[VIEW_uTexture] = tex.view;
475 bindings.samplers[SMP_uSampler] = gXDino_gfxSampler;
476 sg_apply_bindings(&bindings);
477
478 double rotationRadians = rotation * (std::numbers::pi / 180.0);
479 XDino_Linux_Uniforms u;
480 u.half_vp_size_x = gXDino_renderSize.x / 2.f;
481 u.half_vp_size_y = gXDino_renderSize.y / 2.f;
482 u.tex_size_x = static_cast<float>(tex.width);
483 u.tex_size_y = static_cast<float>(tex.height);
484 u.offset_x = static_cast<float>(translation.x);
485 u.offset_y = static_cast<float>(translation.y);
486 u.rot_cos = static_cast<float>(std::cos(rotationRadians));
487 u.rot_sin = static_cast<float>(std::sin(rotationRadians));
488 u.scale = static_cast<float>(scale);
489
490 sg_apply_uniforms(UB_u, {&u, sizeof(u)});
491 sg_draw(0, vbuf.count, 1);
492}
